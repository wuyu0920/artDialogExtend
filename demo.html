信息提示

	qhwDialog.popMsg(type,msg); 
	
	type取值：w || s || f
		w //warning 警告	
		s //success 成功
		f //fail 失败
		
	msg:弹窗的文字信息
	
	eg:qhwDialog.popMsg('s','你成功了')
	
	
	
信息提示(+按钮)

	qhwDialog.alert(type,msg,okFn); 
	
	type取值：w || s || f
		w //warning 警告	
		s //success 成功
		f //fail 失败
		
	msg:弹窗的文字信息
	
	okFn:确定按钮的回调方法
	
	eg:qhwDialog.alert('s','你成功了',function(){}) 或  qhwDialog.alert('s','你成功了',callBack)
	
	function callBack(){}
	
	
confirm

	qhwDialog.confirm(msg,okFn,cancelFn); 
		
	msg:弹窗的文字信息
	
	okFn:确定按钮的回调方法
	
	cancelFn:取消按钮的回调方法
	
	eg:qhwDialog.confirm('你想好了吗',aa,bb)
	
	function aa(){}
	function bb(){}
	
	
	
open新页面

	qhwDialog.open(url,showCallBack);  
	
	url:页面url地址
	
	showCallBack:回调方法
	
	eg:qhwDialog.open('page.htm',cc)
	
	function cc(){}

关闭弹窗

	qhwDialog.close() //关闭扩展方法出来的弹窗
	
	qhwDialog.closeAll()//关闭artDialog所有弹窗
